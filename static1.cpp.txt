#include <iostream>

void staticMemory()
{
    static int a = 0; // Статическая локальная переменная 'a', инициализированная 0. Она сохраняет свое значение между вызовами функции.
    std::cout << "a: " << a << std::endl; // Вывод текущего значения 'a'.
    a += 1; // Увеличение значения 'a' на 1.
}

int main()
{
    staticMemory(); // Первый вызов функции, 'a' увеличивается с 0 до 1.
    staticMemory(); // Второй вызов функции, 'a' увеличивается с 1 до 2.
    staticMemory(); // Третий вызов функции, 'a' увеличивается с 2 до 3.
    return 0;
}
