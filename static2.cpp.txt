#include <iostream>

int a = 0; // Глобальная статическая переменная 'a', инициализированная 0.

void staticMemory()
{
    std::cout << "a: " << a << std::endl; // Вывод текущего значения глобальной переменной 'a'.
    a += 1; // Увеличение значения глобальной переменной 'a' на 1.
}

int main()
{
    staticMemory(); // Первый вызов функции, 'a' увеличивается с 0 до 1.
    staticMemory(); // Второй вызов функции, 'a' увеличивается с 1 до 2.
    staticMemory(); // Третий вызов функции, 'a' увеличивается с 2 до 3.
    return 0;
}
